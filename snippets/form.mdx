export const Form = ({ src, width="500", height="1000" }) => {
  const maxWidth = `${parseInt(width, 10) || 500}px`;
  return <iframe 
        className="form"
        src={src}
        height={height}
        style={{ width: '100%', maxWidth }}
        frameborder="0"
        allowfullscreen>
    </iframe>
};
