Options for getPosts method

## Properties

<ParamField name="sinceId" type="string">
The minimum Post ID to be included in the result set. This parameter takes precedence over start_time if both are specified.

</ParamField>

<ParamField name="untilId" type="string">
The maximum Post ID to be included in the result set. This parameter takes precedence over end_time if both are specified.

</ParamField>

<ParamField name="maxResults" type="number">
The maximum number of results.

</ParamField>

<ParamField name="paginationToken" type="string">
This parameter is used to get the next 'page' of results.

</ParamField>

<ParamField name="exclude" type="string[]">
The set of entities to exclude (e.g. 'replies' or 'retweets').

</ParamField>

<ParamField name="startTime" type="string">
YYYY-MM-DDTHH:mm:ssZ. The earliest UTC timestamp from which the Posts will be provided. The since_id parameter takes precedence if it is also specified.

</ParamField>

<ParamField name="endTime" type="string">
YYYY-MM-DDTHH:mm:ssZ. The latest UTC timestamp to which the Posts will be provided. The until_id parameter takes precedence if it is also specified.

</ParamField>

<ParamField name="tweetfields" type="string[]">
A comma separated list of Tweet fields to display.

</ParamField>

<ParamField name="expansions" type="string[]">
A comma separated list of fields to expand.

</ParamField>

<ParamField name="mediafields" type="string[]">
A comma separated list of Media fields to display.

</ParamField>

<ParamField name="pollfields" type="string[]">
A comma separated list of Poll fields to display.

</ParamField>

<ParamField name="userfields" type="string[]">
A comma separated list of User fields to display.

</ParamField>

<ParamField name="placefields" type="string[]">
A comma separated list of Place fields to display.

</ParamField>

<ParamField name="requestOptions" type="RequestOptions">
Additional request options

</ParamField>
